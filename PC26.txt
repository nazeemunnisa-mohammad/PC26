import 'package:flutter/material.dart';
import 'package:shared_preferences/shared_preferences.dart';

void main() {
  runApp(MaterialApp(home: AlertDialogFormExample()));
}

class AlertDialogFormExample extends StatefulWidget {
  @override
  _AlertDialogFormExampleState createState() => _AlertDialogFormExampleState();
}

class _AlertDialogFormExampleState extends State<AlertDialogFormExample> {
  final _formKey = GlobalKey<FormState>();
  String name = '';
  String lastName = '';
  String email = '';
  String phone = '';

  @override
  void initState() {
    super.initState();
    _loadData(); // Load data on startup
  }

  // Load saved data from shared preferences
  void _loadData() async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    setState(() {
      name = prefs.getString('name') ?? '';
      lastName = prefs.getString('lastName') ?? '';
      email = prefs.getString('email') ?? '';
      phone = prefs.getString('phone') ?? '';
    });
  }

  // Save data to shared preferences
  void _saveData() async {
    if (_formKey.currentState!.validate()) {
      _formKey.currentState!.save();
      SharedPreferences prefs = await SharedPreferences.getInstance();
      await prefs.setString('name', name);
      await prefs.setString('lastName', lastName);
      await prefs.setString('email', email);
      await prefs.setString('phone', phone);
      _loadData(); // Refresh UI after saving
      Navigator.of(context).pop(); // Close dialog
    }
  }

  // Show the AlertDialog
  void _showFormDialog() {
    showDialog(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: Text("Enter Your Details"),
          content: Form(
            key: _formKey,
            child: SingleChildScrollView(
              child: Column(
                children: [
                  TextFormField(
                    decoration: InputDecoration(labelText: 'First Name'),
                    validator: (value) =>
                        value!.isEmpty ? 'Enter first name' : null,
                    onSaved: (value) => name = value!,
                  ),
                  TextFormField(
                    decoration: InputDecoration(labelText: 'Last Name'),
                    validator: (value) =>
                        value!.isEmpty ? 'Enter last name' : null,
                    onSaved: (value) => lastName = value!,
                  ),
                  TextFormField(
                    decoration: InputDecoration(labelText: 'Email'),
                    validator: (value) =>
                        value!.contains('@') ? null : 'Enter valid email',
                    onSaved: (value) => email = value!,
                  ),
                  TextFormField(
                    decoration: InputDecoration(labelText: 'Phone Number'),
                    keyboardType: TextInputType.phone,
                    validator: (value) =>
                        value!.length == 10 ? null : 'Enter 10-digit number',
                    onSaved: (value) => phone = value!,
                  ),
                ],
              ),
            ),
          ),
          actions: [
            ElevatedButton(
              onPressed: _saveData,
              child: Text("Submit"),
            ),
          ],
        );
      },
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("AlertDialog Form Example")),
      body: Padding(
        padding: const EdgeInsets.all(20.0),
        child: Column(
          children: [
            ElevatedButton(
              onPressed: _showFormDialog,
              child: Text("Open Form"),
            ),
            SizedBox(height: 30),
            Text("Saved Data:", style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold)),
            SizedBox(height: 10),
            Text("First Name: $name", style: TextStyle(fontSize: 18)),
            Text("Last Name: $lastName", style: TextStyle(fontSize: 18)),
            Text("Email: $email", style: TextStyle(fontSize: 18)),
            Text("Phone: $phone", style: TextStyle(fontSize: 18)),
          ],
        ),
      ),
    );
  }
}
